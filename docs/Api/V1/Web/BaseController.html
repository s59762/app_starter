def current_cart
  @current_cart ||= Cart.find_by(id: session[:cart_id])

  raise CartFetchFailureException, 'No cart info in session of current user. This API is for browser use only.' unless @current_cart

  @current_cart
end

def product_index_extra_options
  if signed_in_as?('Admin')
    {
      include: [:master],
      each_serializer: ProductSerializer::Detail
    }
  else
    {
      include: [:master]
    }
  end
end
