def save
  ::ActiveRecord::Base.transaction do
    model.update stock: model.stock + normalized_quantity
    model.product.create_activity key: "#{action}_stock",
                                  owner: current_admin,
                                  params: {
                                    quantity: quantity,
                                    variant_id: model.id,
                                    reason: reason
                                  }
  end
end
