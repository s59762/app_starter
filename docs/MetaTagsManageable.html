def autoload_page_title_for_meta_tags
  title = [
    I18n.t("page_titles.#{controller_path.sub('/', '.')}.main_title"),
    I18n.t("page_titles.#{controller_path.sub('/', '.')}.#{action_name}")
  ]

  set_meta_tags title: title
end
